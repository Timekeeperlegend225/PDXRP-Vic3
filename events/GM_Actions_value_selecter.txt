namespace = value_selection

value_selection.1 = {
	type = country_event
	placement = ROOT
	title = value_selection.1.t
	desc = value_selection.1.u

	event_image = {
		texture = "gfx/event_pictures/void.dds"
	} 

	duration = 1
	
	trigger = {
		#triggered by diplo action
	}

	immediate = {
		random_country = {
			limit = {
				has_variable = actionee
			}
			save_scope_as = recipient
		}
	}
	
	#return to GM tool
	
	option = {
		name = value_selection.1.a
		default_option = yes
		trigger_event = {
			id = gm_actions.1
			days = 0
			popup = yes
		}
	}
	

	#Reset
	
	option = {
		name = value_selection.1.b
		set_global_variable = {
			name = gm_variable
			value = 0
		}
		trigger_event = {
			id = value_selection.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = value_selection.1.c
		change_global_variable = {

		}

		}

		}

		}
	}
	
	option = {

			name = gm_variable

		}
