namespace = gm_state_actions

gm_state_actions.1 = {
	type = country_event
	placement = ROOT
	title = gm_state_actions.1.t
	desc = gm_state_actions.1.u

	event_image = {
		texture = "gfx/event_pictures/void.dds"
	} 

	duration = 1

	immediate = {
		random_country = {
			limit = {
				has_modifier = GM_target
			}
			save_scope_as = scope_GM_target
		}
	}
	
	trigger = {
		#triggered by diplo action
	}
	
	#GM options
	
	option = {
		name = gm_state_actions.1.a
		default_option = yes
		remove_global_variable = GM_taking_action
		scope:scope_target_state = {
			remove_modifier = GM_working_on_state_mod
		}
	}
	
	option = {
		name = gm_state_actions.1.b
		trigger = {
			scope:scope_GM_target = {
				NOT = {
					has_claim = scope:scope_target_state 
				}
			}
		}
		
		scope:scope_target_state = {
			state_region = {
				add_claim = scope:scope_GM_target
			}
		}
		trigger_event = {
			id = gm_state_actions.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_state_actions.1.c
		trigger = {
			scope:scope_GM_target = {
				has_claim = scope:scope_target_state 
			}
		}
		
		scope:scope_target_state = {
			state_region = {
				remove_claim = scope:scope_GM_target
			}
		}
		trigger_event = {
			id = gm_state_actions.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_state_actions.1.d
		trigger = {
			scope:scope_target_state  = {
				is_incorporated = no
			}
		}
		
		scope:scope_target_state = {
			set_state_type = incorporated
		}
		trigger_event = {
			id = gm_state_actions.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_state_actions.1.e
		trigger = {
			scope:scope_target_state  = {
				is_incorporated = yes
			}
		}
		
		scope:scope_target_state = {
			set_state_type = unincorporated
		}
		trigger_event = {
			id = gm_state_actions.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_state_actions.1.f
		trigger = {
			scope:scope_target_state = {
				NOT = {
					owner = scope:scope_GM_target
				}
			}
		}
		scope:scope_target_state = {
			set_state_owner = scope:scope_GM_target
		}
		trigger_event = {
			id = gm_state_actions.1
			days = 0
			popup = yes
		}
	}
	
	# Devastation
	option = {
		name = gm_state_actions.3.t
		trigger_event = {
			id = gm_state_actions.3
			days = 0
			popup = yes
		}
	}
}
