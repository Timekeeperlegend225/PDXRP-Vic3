country_can_form_power_bloc = {
	AND = {
		trigger_if = {
			limit = {
				is_country_type = decentralized
			}
			NOT = { is_country_type = decentralized }
		}
		trigger_if = {
			limit = {
				is_revolutionary = yes
			}
			is_revolutionary = no
		}
		country_rank >= rank_value:minor_power
		NOT = { has_law_or_variant = law_type:law_isolationism }
	}
}

power_bloc_is_weak = {
	always = no
