global_springtime_decision = {
	is_shown = {
        has_global_variable = age_of_springtime_gm
		NOT = {
			capital = {
				OR = {
					region = sr:region_england
					region = sr:region_north_sea_coast
					#region = sr:region_iberia #remove this when carlist flavor
					region = sr:region_france
					region = sr:region_occitania
					region = sr:region_rhine
					region = sr:region_italy
					region = sr:region_balkans
					region = sr:region_north_germany
					region = sr:region_south_germany
					region = sr:region_baltic
					region = sr:region_danubia
					region = sr:region_poland
					region = sr:region_russia
					#region = sr:region_dnieper
					region = sr:region_iberia
					region = sr:region_anatolia
				}
			}
		}
		NOT = { has_variable = global_springtime }
		game_date < 1852.1.1
	}

	possible = {
		custom_tooltip = {
			text = "The Age of People started less than 3 years ago (Before 1852)"
			has_global_variable = age_of_springtime_gm
		}
		NOT = { has_law = law_type:law_isolationism }
	}
	when_taken = {
		set_variable = global_springtime
		add_journal_entry = { type = je_springtime_of_the_peoples_global }
		ordered_interest_group = {
			limit = {
				leader = {
					has_ideology = ideology:ideology_radical
				}
			}
			order_by = ig_clout
			leader = {
				save_scope_as = radical_leader_scope
			}
			add_modifier = {
				name = 1848_popular_radical
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		if = {
			limit = {
				any_interest_group = {
					is_marginal = no
					can_have_radical_leader = yes
					is_in_government = no
				}
			}
			random_interest_group = {
				limit = {
					is_marginal = no
					can_have_radical_leader = yes
					is_in_government = no
					leader = {
						NOT = { has_modifier = 1848_popular_radical }
					}
				}
				leader = {
					set_ideology = ideology:ideology_radical
					add_modifier = {
						name = 1848_popular_radical
						days = normal_modifier_time
						is_decaying = yes
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_radical
					}
				}
			}
			create_political_movement = { type = movement_radical }
		}
		add_radicals = {
			value = 0.123
		}
		
	}
	ai_chance = {
		value = 0
	}

}

nationalist_claim = {
	is_shown = {
        has_global_variable = age_of_springtime_gm
		any_state_region = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
				NOT = { owner = ROOT }
				NOT = { has_claim_by = ROOT }
			}
		}
	}

	possible = {
		has_technology_researched = nationalism
		any_state_region = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
				NOT = { owner = ROOT }
				NOT = { owner = { has_subject_relation_with = ROOT } 
				NOT = { has_claim_by = ROOT }
			}
		}
	}

	when_taken = {
		every_state_region = {
			limit = {
				any_scope_state = {
					is_homeland_of_country_cultures = ROOT
					NOT = { owner = ROOT }
					NOT = { owner = { has_subject_relation_with = ROOT } 
					NOT = { has_claim_by = ROOT }
				}
			}
			add_claim = ROOT
		}
	}
	ai_chance = {
		value = 0
	}

}
