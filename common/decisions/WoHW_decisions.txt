			activate_law = law_type:law_religious_schools
			activate_law = law_type:law_charitable_health_system
			activate_law = law_type:law_censorship
			activate_law = law_type:law_no_workers_rights
			activate_law = law_type:law_child_labor_allowed
			activate_law = law_type:law_no_womens_rights
			activate_law = law_type:law_no_social_security
			activate_law = law_type:law_closed_borders
			activate_law = law_type:law_slave_trade
		}
		remove_modifier = game_master
		play_as = c:EMB

			add_modifier = game_master
			add_technology_researched = ember_empire_starting_boost

					imperious
					cruel
					expert_offensive_planner
					experienced_colonial_administrator

				}
			}
		}
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
		every_scope_pop = {
				change_poptype = pop_type:slaves
			}
			create_pop = {
				pop_type = aristocrats

			}
			create_pop = {
				pop_type = soldiers

			}
			state_region = {
				add_homeland = cu:dracaryans
				add_homeland = cu:dragokins
				add_arable_land = 300
			}
		}
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
			create_building = {
				building = building_hole_emb
				level = 20
			}
		}
	}

		base = 0

chakravadah_decision = {
	is_shown = {
		exists = c:PNR
		NOT = { country_has_state_religion = rel:chakravadah }
		has_variable = chakravadah
	}

	possible = {
		ig:ig_devout = {
			is_marginal = no
		}
		has_technology_researched = forced_conversion
		has_variable = chakravadah
	}

	when_taken = {
		set_state_religion = rel:chakravadah
		state_religion_switch_effect = yes
		ig:ig_devout = {
			hidden_effect = { set_interest_group_name = ig_chakravadah }
			remove_ideology = { 
				ideology_buddhist_moralist
				ideology_papal_moralist
				ideology_confucian
				ideology_shinto_moralist
				ideology_hindu_moralist_british_india
				ideology_sikh_moralist
				ideology_heavenly_kingdom_theocratic
				ideology_moralist
			}				
			add_ideology = ideology_hindu_moralist
			set_ig_trait = ig_trait:ig_trait_haibutsu_kishaku
			set_ig_trait = ig_trait:ig_trait_yeshivot
			set_ig_trait = ig_trait:ig_trait_the_best_revenge
			leader ?= {
				change_character_religion = rel:chakravadah
			}
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:chakravadah
				value = 0.4
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:chakravadah
				value = 0.2
			}
		}
	}

	ai_chance = {
		base = 10
	}
}
