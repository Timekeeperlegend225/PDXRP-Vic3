become_EMB = {
	is_shown = {
		has_modifier = game_master
		exist = c:EMB
	}
	possible = {
		always = yes
    }
	
	when_taken = {
		create_country = {
			tag = EMB
			origin = c:GBR
			state = s:STATE_CANARY_ISLANDS.region_state:SPA
		}
		c:EMB ={
			activate_law = law_type:law_monarchy
			activate_law = law_type:law_oligarchy
			activate_law = law_type:law_national_supremacy
			activate_law = law_type:law_state_religion
			activate_law = law_type:law_hereditary_bureaucrats
			activate_law = law_type:law_mass_conscription
			activate_law = law_type:law_national_guard
			activate_law = law_type:law_interventionism
			activate_law = law_type:law_isolationism
			activate_law = law_type:law_per_capita_based_taxation
			activate_law = law_type:law_tenant_farmers
			activate_law = law_type:law_colonial_exploitation
			activate_law = law_type:law_militarized_police
			activate_law = law_type:law_religious_schools
			activate_law = law_type:law_charitable_health_system
			activate_law = law_type:law_censorship
			activate_law = law_type:law_no_workers_rights
			activate_law = law_type:law_child_labor_allowed
			activate_law = law_type:law_no_womens_rights
			activate_law = law_type:law_no_social_security
			activate_law = law_type:law_closed_borders
			activate_law = law_type:law_slave_trade
		}
		remove_modifier = game_master
		play_as = c:EMB
		add_modifier = game_master
		add_technology_researched = ember_empire_starting_boost
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
		every_scope_pop = {
				change_poptype = pop_type:slaves
			}
			create_pop = {
				pop_type = aristocrats
				culture = dracaryans
				religion = nizoism
				size = 4000000
			}
			create_pop = {
				pop_type = soldiers
				culture = dragokins
				religion = nizoism
				qualifications = 40
				size = 12000000
			}
			state_region = {
				add_homeland = cu:dracaryans
				add_homeland = cu:dragokins
			}
		}
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
			create_building = {
				building = building_hole_emb
				level = 20
			}
		}
	}

	ai_chance = {
		base = 0	
	}
}

go_to_war_EMB = {
	is_shown = {
		this = c:EMB
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
			state_has_building_type_levels = {  
				target = bt:building_barracks  
				value < 100
			}
		}
	}
	possible = {
    }
	
	when_taken = {
		s:STATE_CANARY_ISLANDS.region_state:EMB = {
			create_building = {
				building = building_barracks
				level = 200
			} 
			create_building = {
				building = building_naval_base
				level = 100
			} 
		}
	}

	ai_chance = {
		base = 0	
	}
}
