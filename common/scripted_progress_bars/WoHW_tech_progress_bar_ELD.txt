je_WhOW_tech_ELD_pro_progress_bar = {
	name = "je_WhOW_tech_ELD_pro_progress_bar"
	desc = "je_WhOW_tech_ELD_pro_progress_bar_desc"

	default_green = yes

	weekly_progress = {
		if = {
			limit = {
				has_modifier = ELD_studies
			}
			add = 1
		}
		if = {
			limit = {
				has_modifier = studying_all_OPM
			}
			add = 0.2
		}
		else_if = {
			limit = {
				has_modifier = focus_on_studying_ELD
			}
			add = 1
		}
		else_if = {
			limit = {
				OR = {
					has_modifier = focus_on_studying_EMB
					has_modifier = focus_on_studying_PNR
					has_modifier = focus_on_studying_ALA
				}
			}
			add = -0.1
		}
		if = {
			limit = {
				is_in_same_power_bloc = c:ELD
			}
			add = 0.8
		}
		else_if = {
			limit = {
				OR = {
					is_in_same_power_bloc = c:EMB
					is_in_same_power_bloc = c:PNR
					is_in_same_power_bloc = c:ALA
				}
			}
			add = -0.2
		}
		if = {
			limit = {
				is_subject_of = c:ELD
			}
			add = 0.5
		}
		else_if = {
			limit = {
				OR = {
					is_subject_of = c:EMB
					is_subject_of = c:PNR
					is_subject_of = c:ALA
				}
			}
			add = -0.1
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ELD
					type = superior_embargo
				}
			}
			add = -0.3
		}
		if = {
			limit = {
				has_diplomatic_pact = {

				}

			limit = {
				has_diplomatic_pact = {

		if = {

					who = c:ELD
					type = alliance
				}
			}
			add = 0.5
		}
		else_if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ELD
					type = defensive_pact
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				C:ELD.relations:root >= 50
			}
			add = 0.15
		}
		if = {
			limit = {
				has_modifier = ELD_focus_prod
			}
			add = 1
		}
		else_if = {
			limit = {
				OR = {
					has_modifier = ELD_focus_mil
					has_modifier = ELD_focus_soc
				}
			}
			add = -0.4
		}
		if = {
			limit = {
				has_war_with = c:ELD
			}
			add = 1
		}
		if = {
			limit = {
				NOT = { has_technology_researched = forest_harmony }
				OR = {
					has_technology_researched = elven_first_aid
					has_technology_researched = precision_archery
					has_technology_researched = efficient_logistics
					has_technology_researched = elven_healing
				}
			}
			add = 0.1
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = natural_culture }
				OR = {
					has_technology_researched = precision_archery
					has_technology_researched = efficient_logistics
					has_technology_researched = elven_healing
				}
			}
			add = 0.1
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = alchemy }
				OR = {
					has_technology_researched = efficient_logistics
					has_technology_researched = elven_healing
				}
			}
			add = 0.1
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = evlven_smithing }
				has_technology_researched = elven_healing
			}
			add = 0.1
		}
		if = {
			limit = {

				OR = {
					has_technology_researched = elven_aura
					has_technology_researched = plantiri_tech

		else_if = {

		else_if = {

			add = 0.1

		else_if = {

		}
		else_if = {
