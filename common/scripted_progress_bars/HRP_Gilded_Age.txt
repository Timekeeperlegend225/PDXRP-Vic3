HRP_GA_national_speculation = {
	name = "HRP_GA_national_speculation_bar"
	desc = "HRP_GA_national_speculation_desc"

	default_green = yes

	monthly_progress = {
		add = {
			desc = "Base"
			value = 3
		}
		if = {
			limit = {
				has_law = law_type:law_interventionism
			}
			add = {
				desc = "From Interventionism"
				value = 1
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_agrarianism
			}
			add = {
				desc = "From Agrarianism"
				value = 0.5
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_industry_banned
			}
			add = {
				desc = "From Industry Banned"
				value = -0.5
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_laissez_faire
			}
			add = {
				desc = "From Laissez Faire"
				value = 2
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_command_economy
			}
			add = {
				desc = "From Command Economy"
				value = -1
			}
		}
		if = {
			limit = {
				has_law = law_type:law_mercantilism
			}
			add = {
				desc = "From Mercantilism"
				value = 0.5
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_free_trade
			}
			add = {
				desc = "From Free Trade"
				value = 1
			}
		}
		else_if = {
			limit = {
				has_law = law_type:law_isolationism
			}
			add = {
				desc = "From Isolationism"
				value = -1
			}
		}
		add = {
			desc = "From Reliance on Trade"
			add = market.market_trade_reliance
			multiply  = 2
		}
		if = {
			limit = {
				has_technology_researched = joint_stock_companies
			}
			add = {
				desc = "From Joiny-Stock Compagnies"
				value = 0.1
			}
		}
		if = {
			limit = {
				has_technology_researched = mutual_funds
			}
			add = {
				desc = "From Mutual Funds"
				value = 0.2
			}
		}
		if = {
			limit = {
				has_technology_researched = investment_banks
			}
			add = {
				desc = "From Investment Bank"
				value = 0.5
			}
		}
		if = {
			limit = {
				has_technology_researched = international_exchange_standards
			}
			add = {
				desc = "From International Exchange Standart"
				value = 1
			}
		}
		add = {
			desc = "from RP"
			add = var:GA_rp_national
			multiply  = 10
		}
		subtract = {
			desc = "Balance"
			add = "scope:journal_entry.scripted_bar_progress(HRP_GA_national_speculation)" 
			divide  = 10
		}
	}

	start_value = 0
	min_value = 0
	max_value = 100

}
