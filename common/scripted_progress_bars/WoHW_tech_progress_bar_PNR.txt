je_WhOW_tech_PNR_pro_progress_bar = {
	name = "je_WhOW_tech_PNR_pro_progress_bar"
	desc = "je_WhOW_tech_PNR_pro_progress_bar_desc"

	default_green = yes

	weekly_progress = {
		if = {
			limit = {
				has_modifier = PNR_studies
			}
			add = 1
		}
		if = {
			limit = {
				has_modifier = studying_all_OPM
			}
			add = 0.5
		}
		else_if = {
			limit = {
				has_modifier = focus_on_studying_PNR
			}
			add = 2
		}
		else_if = {
			limit = {
				OR = {
					has_modifier = focus_on_studying_EMB
					has_modifier = focus_on_studying_ALA
					has_modifier = focus_on_studying_ELD
					has_modifier = focus_on_studying_VAM
				}
			}
			add = -0.25
		}
		if = {
			limit = {
				is_in_same_power_bloc = c:PNR
			}
			add = 1
		}
		if = {
			limit = {
				is_subject_of = c:PNR
			}
			add = 0.5
		}
		else_if = {
			limit = {
				OR = {
					is_subject_of = c:EMB
					is_subject_of = c:ALA
					is_subject_of = c:ELD
					is_subject_of = c:VAM
				}
			}
			add = -0.1
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:PNR
					type = trade_agreement
				}
			}
			add = 0.25
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:PNR
					type = foreign_investment_rights
				}
			}
			add = 0.25
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:PNR
					type = alliance
				}
			}
			add = 0.5
		}
		else_if = {
			limit = {
				has_diplomatic_pact = {
					who = c:PNR
					type = defensive_pact
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				C:PNR.relations:root >= 50
			}
			add = 0.25
		}
		if = {
			limit = {
				has_modifier = PNR_focus_prod
			}
			add = 1
		}
		else_if = {
			limit = {
				OR = {
					has_modifier = PNR_focus_mil
					has_modifier = PNR_focus_soc
				}
			}
			add = -0.25
		}
		if = {
			limit = {
				has_war_with = c:PNR
			}
			add = 1
		}
		if = {
			limit = {
				NOT = { has_technology_researched = rural_society }
				OR = {
					has_technology_researched = snakeship_tech
					has_technology_researched = beast_cavalry_tech
					has_technology_researched = goldspear_infantry_tech
					has_technology_researched = guerilla_warfare
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_tech }
				OR = {
					has_technology_researched = beast_cavalry_tech
					has_technology_researched = goldspear_infantry_tech
					has_technology_researched = guerilla_warfare
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_fertilizer }
				OR = {
					has_technology_researched = goldspear_infantry_tech
					has_technology_researched = guerilla_warfare
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_breeding }
				has_technology_researched = guerilla_warfare
			}
			add = 0.25
		}
		if = {
			limit = {
				NOT = { has_technology_researched = rural_society }
				OR = {
					has_technology_researched = religious_bureaucracy
					has_technology_researched = forced_conversion
					has_technology_researched = chakravadach_enlightmen
					has_technology_researched = baby_and_work
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_tech }
				OR = {
					has_technology_researched = forced_conversion
					has_technology_researched = chakravadach_enlightmen
					has_technology_researched = baby_and_work
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_fertilizer }
				OR = {
					has_technology_researched = chakravadach_enlightmen
					has_technology_researched = baby_and_work
				}
			}
			add = 0.25
		}
		else_if = {
			limit = {
				NOT = { has_technology_researched = nagan_cattle_breeding }
				has_technology_researched = baby_and_work
			}
			add = 0.25
		}
		if = {
			limit = {
				NOT = { has_technology_researched = rural_society }
			}
			multiply = 2
		}
		else_if = {
			limit = {
				has_technology_researched = nagan_cattle_tech
			}
			multiply = 0.5
		}
		else_if = {
			limit = {
				has_technology_researched = nagan_cattle_fertilizer
			}
			multiply = 0.25
		}
	}

	start_value = 0
	min_value = 0
	max_value = 1000

}
