je_springtime_of_the_peoples = {
    icon = "gfx/interface/icons/event_icons/waving_flag.dds"

    group = je_group_historical_content

    immediate = {
		if = {
			limit = {
				var:tempnationalstab < 150
			}
			scope:journal_entry = { 
				add_modifier = {
					name = modifier_springtime_je
					days = normal_modifier_time
					is_decaying = no
				}
			}
		}
		else_if = {
			limit = {
				var:tempnationalstab < 350
			}
			scope:journal_entry = { 
				add_modifier = {
					name = modifier_springtime_je
					multiplier = 0.7
					days = normal_modifier_time
					is_decaying = no
				}
			}
		}
		else = {
			scope:journal_entry = { 
				add_modifier = {
					name = modifier_springtime_je
					multiplier = 0.4
					days = normal_modifier_time
					is_decaying = no
				}
			}
		}
		trigger_event = { id = 1848.4 days = 0 }
		set_variable = {
			name = springtime_delay_var
			months = 12
		}
	}

    modifiers_while_active = {
	}
    
    complete = {
		custom_tooltip = {
			text = "Spring time of the Peoples has lasted at least a year."
			NOT = { has_variable = springtime_delay_var }
		}
		is_at_war = no
		has_revolution = no
		is_revolutionary = no
		has_insurrectionary_interest_groups = no
		NOT = {
			any_interest_group = {
				leader = { has_ideology = ideology:ideology_radical }
				is_marginal = no
			}
		}
	}

	on_complete = {
		set_variable = {
			name = completed_peoples_stringtime
		}
		set_variable = {
			name = completed_peoples_stringtime_recent
			days = short_modifier_time
		}
		if = {
			limit = {
				NOT = {
					any_interest_group = {
						leader = { has_ideology = ideology:ideology_radical }
						is_marginal = no
					}
				}
			}
			trigger_event = { id = peoples_springtime.8 }
		}
		add_modifier = {
			name = counter_revolutionary_fervor
			months = 120
			is_decaying = yes
		}
		
	}

	fail = {
		OR = {
			has_revolution = yes
		}
	}
	
	on_fail = {
		set_variable = {
			name = completed_peoples_stringtime
		}
		set_variable = {
			name = completed_peoples_stringtime_recent
			days = short_modifier_time
		}
		trigger_event = { id = peoples_springtime.7 }
		add_modifier = {
			name = modifier_1848_loss
			years = 20	
			is_decaying = yes
		}
		#add_modifier = {
		#	name = post_revolutionary_fervor
		#	years = 20
		#	is_decaying = yes
		#}
		if = {
			limit = {
	       		is_in_power_bloc = yes
	        	is_power_bloc_leader = yes				
			}
			custom_tooltip = {
				text = sick_man_of_europe_lose_power_bloc
				set_country_type = unrecognized   #THIS WILL PROBABLY NOT WORK IN THE FUTURE
				set_country_type = recognized
			}
		}
	}
	
	timeout = 1460 
	
	on_timeout = {
		add_modifier = {
			name = modifier_1848_victory
			months = 60
			multiplier = 0.33
		}
	}

	on_monthly_pulse = {
		events = {
			1848.5
		}
		random_events = {
			100 = 0
			10 = peoples_springtime.3
			10 = peoples_springtime.4
			10 = peoples_springtime.5
			10 = austria_monarchy.1
			20 = dual_monarchy.3
		}
	}

	#invalid = {
	#	custom_tooltip = {
	#		text = je_springtime_of_the_peoples.t2
	#		has_global_variable = peoples_springtime_happened
	#	}
	#}

	weight = 10000
	should_be_pinned_by_default = yes

	transferable = no
	can_revolution_inherit = no
}
