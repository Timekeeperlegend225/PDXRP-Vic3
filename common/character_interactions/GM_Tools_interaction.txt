GM_kill_char = {
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/exile_character"

	potential = {
		owner ?= {
			has_modifier = game_master
		}
	}

	show_requirements = yes

	possible = {
		always = yes
	}

	effect = {
		ordered_country = {
			limit = {
				has_modifier = GM_target
			}
			save_scope_as = owner_scope
			remove_modifier = GM_target
		}
		transfer_character = scope:owner_scope
		kill_character = yes
	}
}
