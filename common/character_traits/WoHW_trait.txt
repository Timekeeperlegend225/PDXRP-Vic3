acolyte  = {
 	type = skill
	texture = gfx/interface/icons/character_trait_icons/pious.dds

	character_modifier = {
		character_health_add = 0.5
		character_popularity_add = 5
	}

	country_modifier = {
		country_influence_mult = 0.05
		country_legitimacy_base_add = 3
		country_prestige_add = 10
	}
	
	interest_group_modifier = {
		interest_group_pol_str_mult = 0.05
		interest_group_pop_attraction_mult = 0.05
	}

	agitator_modifier = {
		political_movement_pop_attraction_mult = 0.10
	}

	command_modifier = {
		character_command_limit_mult = 0.05
		unit_morale_recovery_mult = 0.05
		unit_combat_unit_type_light_artillery_offense_add = 10
	}

 	possible = {
 		owner ?= {
 			has_technology_researched = purification
 		}
 		NOT = { age < define:NCharacters|ADULT_AGE }
 	}

 	weight = {
 		value = 1
		if = {
			limit = {
				has_trait = pious
			}
			multiply = 5
		}
 	}

 	replace = {
		inexperienced
 	}

 	value = 1
}
