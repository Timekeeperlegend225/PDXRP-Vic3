je_WhOW_tech_ALA_focus = {
	name = "je_WhOW_tech_ALA_focus"
	desc = "je_WhOW_tech_ALA_focus_desc"
	
	visible = { always = yes }

	possible = {
		scope:journal_entry = {
			not = {
				custom_tooltip = {
					text = positivist_cooldown_tt
					has_variable = WoHW_focus_var
				}
			}
		}
	}

	effect = {
		add_modifier = focus_on_studying_ALA
		if = {
			limit = {
				has_modifier = focus_on_studying_EMB
			}
			remove_modifier = focus_on_studying_EMB
		}
		if = {
			limit = {
				has_modifier = focus_on_studying_PNR
			}
			remove_modifier = focus_on_studying_PNR
		}
		if = {
			limit = {
				has_modifier = focus_on_studying_ELD
			}
			remove_modifier = focus_on_studying_ELD
		}
		if = {
			limit = {
				has_modifier = focus_on_studying_VAMP
			}
			remove_modifier = focus_on_studying_VAMP
		}
		scope:journal_entry = {
			custom_tooltip = cooldown_year_tt
			set_variable = { 
				name = WoHW_focus_var 
				years = 1 
			}
		}
	}
}
